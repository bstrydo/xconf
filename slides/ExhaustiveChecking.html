enum Weekday {
    Monday,
    Tuesday,
    Wednesday,
    Thursday,
    Friday,
    Saturday,
    Sunday
}

let day_of_the_week = Weekday::Monday;

match day_of_the_week {
    Monday | Tuesday => println!("Early in the week"),
    Wednesday => println!("Down hill from here"),
    Thursday => println!("Party night"),
    Friday => println!("the weekend is here"),
    Saturday | Sunday => println!("Its time to play and braai"),
}
